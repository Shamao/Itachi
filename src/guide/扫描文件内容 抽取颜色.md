const fs = require('fs')

const readDir = function (path) {
  fs.readdir(path, (error, files) => {
    if (!error) {
      files.forEach(file => {
        let newFile = path + '/' + file
        if (fs.existsSync(newFile)) {
          if (fs.existsSync(newFile) && fs.statSync(newFile).isFile()) {
            readFile(newFile)
          } else {
            readDir(newFile)
          }
        } else {
          console.log('文件不存在 : ' + file)
        }
      })
    } else {
      console.log('读取错误：' + error)
    }
  })
}

const readFile = function (path) {
  fs.readFile(path, null, (error, data) => {
    if (!error) {
      const reg = /#[0-9a-zA-Z]{6}/g
      const colors = data.toString().match(reg)
      if (colors != null && colors.length > 0) {
        colors.forEach(color => {
          if (color.toUpperCase() <= '#FFFFFF') {
            colorsG.push(color.toUpperCase())
          }
        })
        flushColor()
      }
    } else {
      console.log('读取错误：' + error)
    }

  })
}


const excludeRepeat = (colors) => {
  const result = []
  colors.forEach((color) => {
    if (result.indexOf(color) === -1) {
      result.push(color)
    }
  })

  return result
}

const flushColor = () => {
  const exportPath = './src/utils/ColorUtils1.js'
  let content = ''
  excludeRepeat(colorsG).sort().forEach((color) => {
    const color1 = color.substring(1, color.length).toUpperCase()
    content += `  C_${color1}: \'${color}\',\n`
  })
  const prefixContent = 'const Colors = {\n'
  let suffixContent = '}\nexport default Colors\n'
  const result = prefixContent + content + suffixContent
  writeFile(exportPath, result)
}

const writeFile = function (path, content) {
  fs.writeFile(path, content, null, (error) => {
    if (error) {
      console.log(error)
    } else {
      console.log('输出成功')
    }
  })
}
